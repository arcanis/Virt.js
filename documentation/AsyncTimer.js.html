import { SerialTimer } from 'virtjs/devices/timers/SerialTimer';

export class AsyncTimer {

    /**
     * An AsyncTimer is an asynchronous timer. You can use it to asynchronously run your emulator. You probably don't want to use it directly: if you just want to run your emulator asynchronously on a browser then {@link AnimationFrameTimer} should fit most of your needs, and if you want to do it on Node then {@link ImmediateTimer} is your go-to timer.
     *
     * @constructor
     * @implements {Timer}
     *
     * @param {object} [options] - The timer options.
     * @param {function} [options.prepare] - The callback that will schedule the next cycle
     * @param {function} [options.cancel] - The callback that will abort the next cycle
     *
     * @see {@link AnimationFrameTimer}
     * @see {@link ImmediateTimer}
     */

    constructor({ prepare, cancel } = { }) {

        if (prepare)
            this.prepare = prepare;

        if (cancel)
            this.cancel = cancel;

        this.loopHandler = null;
        this.timer = new SerialTimer();

    }

    nextTick(callback) {

        return this.timer.nextTick(callback);

    }

    cancelTick(handler) {

        return this.timer.cancelTick(handler);

    }

    start(beginning, ending) {

        if (this.loopHandler !== null)
            return;

        /* eslint-disable no-nested-ternary */

        let loop = beginning && ending ? () => {

            this.loopHandler = this.prepare(loop);

            beginning();
            this.timer.one();
            ending();

        } : beginning ? () => {

            this.loopHandler = this.prepare(loop);

            beginning();
            this.timer.one();

        } : ending ? () => {

            this.loopHandler = this.prepare(loop);

            this.timer.one();
            ending();

        } : () => {

            this.loopHandler = this.prepare(loop);

            this.timer.one();

        };

        /* eslint-enable no-nested-ternary */

        loop();

    }

    stop() {

        if (this.loopHandler !== null)
            return;

        this.cancel(this.loopHandler);
        this.loopHandler = null;

    }

    /**
     * This method should be specialized, either via subclassing, or by passing the proper parameter when instanciating the timer.
     *
     * @protected
     *
     * @type {prepareCallback}
     */

    prepare() {

        throw new Error(`Unimplemented`);

    }

    /**
     * This method should be specialized, either via subclassing, or by passing the proper parameter when instanciating the timer.
     *
     * @protected
     *
     * @type {cancelCallback}
     */

    cancel() {

        throw new Error(`Unimplemented`);

    }

}
